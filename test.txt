def generate_ai_brief(client):

    age = client.get("age", 0)
    occupation = client.get("occupation", "")
    risk = client.get("riskProfile", "Medium")
    goal = client.get("goal", "")
    last = client.get("lastMeeting", "")
    priority = client.get("priority", "Normal")

    focus = []

    # 🧠 1. GOAL inference (if keyword exists)
    if goal:
        if "retirement" in goal.lower():
            focus.append("Discuss retirement income planning")
        elif "education" in goal.lower():
            focus.append("Discuss education funding strategy")
        else:
            focus.append(f"Review financial goal: {goal}")

    # 🧠 2. AGE-based inference (NO keyword needed)
    if age >= 50:
        focus.append("Focus on retirement readiness and wealth preservation")
    elif age >= 30:
        focus.append("Focus on wealth accumulation strategy")
    else:
        focus.append("Focus on early-stage financial planning")

    # 🧠 3. OCCUPATION inference (VERY IMPORTANT fallback)
    if occupation:
        if "business" in occupation.lower():
            focus.append("Review business asset protection strategy")
        elif "engineer" in occupation.lower():
            focus.append("Optimize structured investment planning")
        else:
            focus.append(f"Align financial strategy with profession: {occupation}")

    # 🧠 4. RISK fallback (always works even if missing data)
    if risk == "Conservative":
        product = "Recommend low-risk financial products"
    elif risk == "Aggressive":
        product = "Recommend high-growth investment products"
    else:
        product = "Recommend balanced financial portfolio"

    # 🧠 5. LAST MEETING fallback
    if last:
        focus.append(f"Follow up on previous discussion: {last}")
    else:
        focus.append("Establish initial financial profiling discussion")

    # 🧠 6. PRIORITY fallback
    urgency = "Standard follow-up cycle"
    if priority == "High":
        urgency = "Immediate follow-up required"

    # OUTPUT
    return {
        "focus": focus,
        "product": product,
        "urgency": urgency
    }

// api.js
// This file handles backend API calls.
// For now, it returns demo data if backend is not connected.

const API_URL = "http://127.0.0.1:5000";

async function postData(endpoint, data) {
  try {
    const response = await fetch(`${API_URL}${endpoint}`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify(data)
    });

    if (!response.ok) {
      throw new Error("Backend error");
    }

    return await response.json();
  } catch (error) {
    console.warn("Backend not connected. Using demo data instead.");
    return null;
  }
}

async function apiGenerateBrief(client) {
  const result = await postData("/generate-brief", { client });

  if (result) return result;

  return {
    focus: [
      "Discuss retirement income planning.",
      "Recommend conservative financial products.",
      "Follow up on daughter’s university funding concern.",
      "Suggest ABC Insurance as a suitable partner."
    ]
  };
}

async function apiGenerateSummary(notes) {
  const result = await postData("/generate-summary", { notes });

  if (result) return result;

  return {
    summary: "The client prefers lower-risk investment options and is interested in retirement income planning.",
    goal: "Build a stable retirement income plan while minimizing exposure to high-risk products.",
    actions: [
      "Send retirement product brochure.",
      "Schedule follow-up meeting next Monday.",
      "Introduce ABC Insurance partner."
    ],
    email: "Dear Alice, thank you for meeting today. Based on our discussion, I will prepare suitable low-risk retirement planning options and share them with you shortly."
  };
}

function saveLoggedInUser(user) {
  localStorage.setItem("isLoggedIn", "true");
  localStorage.setItem("userName", user.name);
  localStorage.setItem("userEmail", user.email);
  localStorage.setItem("userRole", user.role);
}

function showLoginError(message) {
  const errorElement =
    document.getElementById("loginError");

  if (errorElement) {
    errorElement.textContent = message;
  }
}

function handleLogin(event) {
  event.preventDefault();

  const emailInput =
    document.getElementById("email");

  const passwordInput =
    document.getElementById("password");

  const email =
    emailInput.value.trim().toLowerCase();

  const password =
    passwordInput.value;

  showLoginError("");

  if (!email || !password) {
    showLoginError(
      "Please enter your email and password."
    );

    return;
  }

  const matchedUser = demoAccounts.find(
    account =>
      account.email.toLowerCase() === email &&
      account.password === password
  );

  if (!matchedUser) {
    showLoginError(
      "Invalid email or password."
    );

    return;
  }

  saveLoggedInUser(matchedUser);

  window.location.replace("dashboard.html");
}

document.addEventListener("DOMContentLoaded", function () {
  const alreadyLoggedIn =
    localStorage.getItem("isLoggedIn") === "true";

  if (alreadyLoggedIn) {
    window.location.replace("dashboard.html");

    return;
  }

  const loginForm =
    document.getElementById("loginForm");

  if (loginForm) {
    loginForm.addEventListener(
      "submit",
      handleLogin
    );
  }
});

  `;
}

}
